Examples

1. Simple Pod creation

kind: Pod
apiVersion: v1
metadata:
        name: testpod1
spec:
        containers:
                - name: c00
                  image: ubuntu
                  command: ["/bin/bash", "-c", "while true; do echo Hello-Siddharth; sleep 5 ; done"]
        restartPolicy: Never      # Defaults to Always       

2. Adding Annotations

kind: Pod
apiVersion: v1
metadata:
        name: testpod2
        annotations:
                description: this is sample annotation
spec:
        containers:
                - name: c00
                  image: ubuntu
                  command: ["/bin/bash", "-c", "while true; do echo Hello-Siddharth; sleep 5 ; done"]
        restartPolicy: Never      # Defaults to Always 

3. Multi Container Pod Environment

kind: Pod
apiVersion: v1
metadata:
        name: testpod3
spec:
        containers:
                - name: c00
                  image: ubuntu
                  command: ["/bin/bash", "-c", "while true; do echo Hello-test1; sleep 5 ; done"]
                - name: c01
                  image: ubuntu
                  command: ["/bin/bash", "-c", "while true; do echo Hello-test2; sleep 5 ; done"]

4. Pods with env variables

kind: Pod
apiVersion: v1 
metadata:
  name: testpod4
spec:
  containers:
    - name: c00
      image: ubuntu
      command: ["/bin/bash", "-c", "while true; do echo Hello-Siddharth; sleep 5 ; done"]
      env:                        # List of environment variables to be used inside the pod
      - name: MYNAME
        value: Sidd


5. Pods with ports

kind: Pod
apiVersion: v1
metadata:
  name: testpod5
spec:					
  containers:
    - name: c00
      image: httpd
      ports:
       - containerPort: 80
